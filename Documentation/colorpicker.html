Color pickr

  1. Forms
  2. Color pickr
OverviewOfficial Website

Colorpickr: Flat, simple, multi-themed, responsive and hackable Color-Picker library.

CSS
<!-- One of the following themes -->
<link rel="stylesheet" href="/static/libs/@simonwep/pickr/themes/classic.min.css"/> <!-- 'classic' theme -->
<link rel="stylesheet" href="/static/libs/@simonwep/pickr/themes/monolith.min.css"/> <!-- 'monolith' theme -->
<link rel="stylesheet" href="/static/libs/@simonwep/pickr/themes/nano.min.css"/> <!-- 'nano' theme -->
Javascript
<!-- Modern colorpicker bundle -->
<script src="/static/libs/@simonwep/pickr/pickr.min.js"></script>
Initjs (Custom js)
<!--  color pickr init js -->
<script src="/static/js/pages/form-pickers.init.js"></script>
Add Package
yarn add @simonwep/pickr --save
Remove Package
yarn remove @simonwep/pickr or you can remove package by removing specific package from package.json

Title Javascript
Classic Demo 
var classicPickrDemo = document.querySelectorAll(".classic-colorpicker");

classicPickrDemo.forEach(function () {
Pickr.create({
    el: ".classic-colorpicker",
    theme: "classic", // or 'monolith', or 'nano'
    default: "#405189",
    swatches: [
        "rgba(244, 67, 54, 1)",
        "rgba(233, 30, 99, 0.95)",
        "rgba(156, 39, 176, 0.9)",
        "rgba(103, 58, 183, 0.85)",
        "rgba(63, 81, 181, 0.8)",
        "rgba(33, 150, 243, 0.75)",
        "rgba(3, 169, 244, 0.7)",
        "rgba(0, 188, 212, 0.7)",
        "rgba(0, 150, 136, 0.75)",
        "rgba(76, 175, 80, 0.8)",
        "rgba(139, 195, 74, 0.85)",
        "rgba(205, 220, 57, 0.9)",
        "rgba(255, 235, 59, 0.95)",
        "rgba(255, 193, 7, 1)",
    ],

    components: {
        // Main components
        preview: true,
        opacity: true,
        hue: true,

        // Input / output Options
        interaction: {
            hex: true,
            rgba: true,
            hsva: true,
            input: true,
            clear: true,
            save: true,
        },
    },
});
});
